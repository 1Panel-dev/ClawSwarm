## 四、API 封装

### 4.1 Axios 实例

```typescript
// src/api/index.ts
import axios from 'axios'
import type { AxiosInstance, AxiosRequestConfig } from 'axios'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/stores/user'
import router from '@/router'

const request: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 10000
})

// 请求拦截器
request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    return config
  },
  (error) => Promise.reject(error)
)

// 响应拦截器
request.interceptors.response.use(
  (response) => {
    const { code, message, data } = response.data
    if (code === 200) {
      return data
    }
    ElMessage.error(message || '请求失败')
    return Promise.reject(new Error(message || '请求失败'))
  },
  (error) => {
    if (error.response?.status === 401) {
      const userStore = useUserStore()
      userStore.logout()
      router.push('/login')
      ElMessage.error('登录已过期，请重新登录')
    } else {
      ElMessage.error(error.message || '网络错误')
    }
    return Promise.reject(error)
  }
)

export default request
```

### 4.2 API 模块示例

```typescript
// src/api/humans.ts
import request from './index'
import type { Human, PageResponse } from '@/types'

export interface HumanCreateParams {
  username: string
  email: string
  password: string
  status?: string
}

export interface HumanListParams {
  page?: number
  page_size?: number
  status?: string
  keyword?: string
}

export const humansApi = {
  // 创建人类账号
  create(data: HumanCreateParams) {
    return request.post('/humans', data)
  },

  // 获取人类账号列表
  list(params: HumanListParams) {
    return request.get<PageResponse<Human>>('/humans', { params })
  },

  // 获取人类账号详情
  detail(id: number) {
    return request.get<Human>(`/humans/${id}`)
  },

  // 更新人类账号
  update(id: number, data: Partial<HumanCreateParams>) {
    return request.put(`/humans/${id}`, data)
  },

  // 禁用人类账号
  disable(id: number) {
    return request.post(`/humans/${id}/disable`)
  },

  // 启用人类账号
  enable(id: number) {
    return request.post(`/humans/${id}/enable`)
  },

  // 删除人类账号
  delete(id: number) {
    return request.delete(`/humans/${id}`)
  }
}
```

## 五、状态管理

### 5.1 用户状态

```typescript
// src/stores/user.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { authApi } from '@/api/auth'

export const useUserStore = defineStore('user', () => {
  const token = ref<string>('')
  const userInfo = ref<any>(null)

  // 登录
  async login(username: string, password: string) {
    const data = await authApi.login({ username, password })
    token.value = data.access_token
    await fetchUserInfo()
  }

  // 获取用户信息
  async fetchUserInfo() {
    userInfo.value = await authApi.me()
  }

  // 退出登录
  logout() {
    token.value = ''
    userInfo.value = null
    localStorage.removeItem('token')
  }

  return { token, userInfo, login, fetchUserInfo, logout }
})
```

```vue
<!-- src/views/Login.vue -->
<template>
  <div class="login-container">
    <el-card class="login-card">
      <h2>1Claw 管理系统</h2>
      <el-form :model="form" :rules="rules" ref="formRef">
        <el-form-item prop="username">
          <el-input v-model="form.username" placeholder="用户名" />
        </el-form-item>
        <el-form-item prop="password">
          <el-input v-model="form.password" type="password" placeholder="密码" />
        </el-form-item>
        <el-form-item>
          <el-button type="primary" @click="handleLogin" :loading="loading">
            登录
          </el-button>
        </el-form-item>
      </el-form>
    </el-card>
  </div>
</template>

<script setup lang="ts">
import { ref, reactive } from 'vue'
import { useRouter } from 'vue-router'
import { useUserStore } from '@/stores/user'
import { ElMessage } from 'element-plus'

const router = useRouter()
const userStore = useUserStore()
const formRef = ref()
const loading = ref(false)

const form = reactive({
  username: '',
  password: ''
})

const rules = {
  username: [{ required: true, message: '请输入用户名', trigger: 'blur' }],
  password: [{ required: true, message: '请输入密码', trigger: 'blur' }]
}

const handleLogin = async () => {
  await formRef.value.validate()
  loading.value = true
  try {
    await userStore.login(form.username, form.password)
    ElMessage.success('登录成功')
    router.push('/')
  } catch (e) {
    // 错误已在拦截器处理
  } finally {
    loading.value = false
  }
}
</script>
```

```vue
<!-- src/views/Humans/List.vue -->
<template>
  <div class="humans-list">
    <el-card>
      <template #header>

              <template #dropdown>
                <el-dropdown-menu>
                  <el-dropdown-item @click="handleView(row)">查看详情</el-dropdown-item>
                  <el-dropdown-item @click="handleEdit(row)">编辑信息</el-dropdown-item>
                  <el-dropdown-item divided @click="handleDisable(row)" v-if="row.status === 'enabled'">
                    禁用账号
                  </el-dropdown-item>
                  <el-dropdown-item @click="handleEnable(row)" v-else>
                    启用账号
                  </el-dropdown-item>
                  <el-dropdown-item divided @click="handleDelete(row)" :disabled="row.status !== 'disabled'">
                    删除账号
                  </el-dropdown-item>
                </el-dropdown-menu>
              </template>
            </el-dropdown>
          </template>
        </el-table-column>
      </el-table>

      <!-- 分页 -->
      <Pagination v-model:page="page" v-model:page-size="pageSize" :total="total" @change="fetchData" />
    </el-card>

    <!-- 添加/编辑弹窗 -->

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { humansApi } from '@/api/humans'
import { ElMessage, ElMessageBox } from 'element-plus'

const loading = ref(false)
const tableData = ref([])
const page = ref(1)
const pageSize = ref(20)
const total = ref(0)
const keyword = ref('')
const status = ref('')
const dialogVisible = ref(false)

      page_size: pageSize.value,

    fetchData()
  } catch (e) {
    // 取消操作
  }

  fetchData()
})
```

```vue
  get: () => props.modelValue,

const handleCancel = () => {
  emit('cancel')
  visible.value = false
```

```typescript
export interface Human {
  id: number
  username: string
  email: string
```
